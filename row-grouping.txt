// Assuming the date format in the column is "18 Jul 2023"
function customDateFilter(dateToSearch, cellValue) {
  // Convert the cellValue to lowercase for case-insensitive comparison
  const lowerCellValue = cellValue.toLowerCase();

  // Convert the search query to lowercase for case-insensitive comparison
  const lowerSearchValue = dateToSearch.toLowerCase();

  // Perform a strict comparison of the full date string
  return lowerCellValue === lowerSearchValue;
}
