var gridApi = gridOptions.api; // Use gridOptions.api to access the GridApi object
var columnApi = gridOptions.columnApi;

var targetColumnFields = ['column1', 'column2']; // Replace with the field names of the target columns

// Get the current column definitions
var columnDefs = columnApi.getColumnDefs();

// Modify the column definitions to set the target columns as row grouping columns
var updatedColumnDefs = columnDefs.map(function(columnDef) {
  if (targetColumnFields.includes(columnDef.field)) {
    return {
      ...columnDef,
      rowGroup: true,
    };
  }
  return columnDef;
});

// Apply the updated column definitions
gridApi.setColumnDefs(updatedColumnDefs);
